131 public boolean isDolbySupported() {
132 //Get the display and the corresponding abilities.
133 Display screen = this.context.getDisplay();
134
135 Display.HdrCapabilities capabilities = screen.getHdrCapabilities();
136
137 //Check if the list of abilities is empty when anything that is not Dolby Vision is filtered.
138 return !(Arrays.stream(capabilities.getSupportedHdrTypes()).
139 filter(x -> x == Display.HdrCapabilities.HDR_TYPE_DOLBY_VISION).
140 boxed().
141 collect(Collectors.toCollection(ArrayList::new)).isEmpty());
142 }

149 private HashMap<Integer, MediaFormat> enumerateTracks() {
150
151 HashMap<Integer, MediaFormat> formats = new HashMap<>();
152
153 //Map all tracks in the file.
154 for (int i = 0; i < extractor.getTrackCount(); i++) {
155 //The key is the track number which is mapped to the format of that track.
156 formats.put(i, extractor.getTrackFormat(i));
157 }
158
159 return formats;
160 }

172 private List<Track> getTracksByType(CodecType type) {
173
174 //For every track in the file
175 return this.enumerateTracks().entrySet().stream().filter(x -> {
176
177 //Ensure the format for that track is present.
178 if (x.getValue() != null) {
179
180 //check if the mime type starts with the desired prefix, video or audio.
181 return x.getValue().getString(MediaFormat.KEY_MIME).toLowerCase(Locale.ROOT).startsWith(type == CodecType.VIDEO ? "video" : "audio");
182
183 } else {
184 //If format is not present.
185 return false;
186 }
187 //Take the output and create tracks from them.
188 }).map(Track::new)
189 .collect(Collectors.toCollection(ArrayList::new));
190 }

220 private Track getPlayableTrack() throws NoCodecException {
221
222 //Get a list of video tracks in the file.
223 List<Track> videoTracks = this.getTracksByType(CodecBuilder.CodecType.VIDEO);
224
225 AtomicBoolean option = new AtomicBoolean(false);
226
227 Log.e("GetPlayableTrack", "getPlayableTrack: Blocking non HEVC Codecs " + this.outputFormat);
228 ArrayList<Track> tracks = videoTracks.parallelStream().filter(x -> {
229
230 //See if there are any Dolby tracks in the file, this filtering only applies to this case.
231 boolean isDolby = videoTracks.parallelStream().anyMatch(f -> f.getFormat().getString(MediaFormat.KEY_MIME).equals(MediaFormat.MIMETYPE_VIDEO_DOLBY_VISION) &&
232 f.getFormat().getInteger(MediaFormat.KEY_PROFILE) == MediaCodecInfo.CodecProfileLevel.DolbyVisionProfileDvheSt);
233
234 if (isDolby) {
235 //Set the decoder for an HEVC output.
236 if (outputFormat.equals(HEVC) || outputFormat.equals(AVC) && !option.get()) {
237 //Eliminate any tracks that are not Dolby Vision.
238 boolean found = x.getFormat().getString(MediaFormat.KEY_MIME).equals(MediaFormat.MIMETYPE_VIDEO_DOLBY_VISION);
239
240 if (found) {
241 option.set(true);
242 }
243
244 return found;
245 }
246 //Set the decoder for Dolby Vision output.
247 else if (outputFormat.equals(DV_ME) && !option.get()) {
248 //Eliminate any tracks that are not HEVC (really Dolby Vision base layer.
249 boolean found = x.getFormat().getString(MediaFormat.KEY_MIME).equals(MediaFormat.MIMETYPE_VIDEO_HEVC);
250
251 if (found) {
252 option.set(true);
253 }
254
255 return found;
256
257 } else {
258 return false;
259 }
260 } else {
261 return true;
262 }
263 })
264 .filter(t -> this.filterCodecByFormat(t.getFormat()) != null)
265 .collect(Collectors.toCollection(ArrayList::new));
266
267 Log.e("GetPlayableTrack", "getPlayableTrack: list " + Arrays.toString(tracks.toArray()));
268
269
270 if (tracks.isEmpty()) {
271 //The device cannot play the file.
272 throw new NoCodecException();
273 } else {
274 return tracks.get(TOP_RESULT);
275 }
276 }

284 private MediaCodecInfo[] nonDolbyKnockout(final boolean dolbyCodecsOnly) {
285 //In order to take advantage of reflection APIs we need an object to operate on.
286 final MediaCodecInfo.CodecProfileLevel reflexiveReference = new MediaCodecInfo.CodecProfileLevel();
287
288 /*
289 This operation reflexively asks the object defined above to return a list of all its
290 declared fields. This returns a list of all class variables, regardless of access level.
291 We then get the name of these and only keep the ones that start with Google's defined prefix
292 for Dolby Vision profiles. The integer set returned has these values in it.
293 */
294 Set<Integer> dolbyProfiles = Arrays.stream(MediaCodecInfo.CodecProfileLevel.class.getDeclaredFields())
295 //Filter for only Dolby Vision profiles.
296 .filter(x -> x.getName().startsWith(REFLEXIVE_DOLBY_PROFILE_LOOKUP))
297 .map(x -> {
298 try {
299 //Map the variable name to its value.
300 return x.getInt(reflexiveReference);
301 } catch (IllegalAccessException e) {
302 e.printStackTrace();
303 }
304 return null;
305 })
306 //Return a list of those values.
307 .collect(Collectors.toSet());
308
309 /*
310 Go through the available codecs on the phone and filter out the ones that do not support
311 at least one of the profiles we found above.
312 */
313 return Arrays.stream(codecInfos).
314 //We are not interested in encoders.
315 filter(x -> !x.isEncoder()).
316 filter(x -> {
317
318 //High level check to see if the codec supports the Dolby Vision mime type.
319 if (Arrays.asList(x.getSupportedTypes()).contains(DOLBY_VISION_TYPE)) {
320
321 //Check for profile support.
322 for (MediaCodecInfo.CodecProfileLevel profLevel : x.getCapabilitiesForType(DOLBY_VISION_TYPE).profileLevels) {
323
324 if (dolbyProfiles.contains(profLevel.profile) && dolbyCodecsOnly) {
325 return true;
326 }
327 }
328 }
329
330 return !dolbyCodecsOnly;
331
332 }).toArray(MediaCodecInfo[]::new);
333

341 public HashMap<Integer, ArrayList<String>> getDolbyCodecMapping() {
342 //Hashmap that we will ultimately return.
343 HashMap<Integer, ArrayList<String>> dolbyProfileToCodecLookup = new HashMap<>();
344
345 //Get a list of Dolby codecs on the phone.
346 Arrays.stream(nonDolbyKnockout(true)).
347 filter(c -> !c.isEncoder()).
348 filter(c -> !c.getCapabilitiesForType(DOLBY_VISION_TYPE).
349 isFeatureSupported(MediaCodecInfo.CodecCapabilities.FEATURE_SecurePlayback)).
350 filter(codec ->
351 Arrays.asList(codec.getSupportedTypes()).contains(DOLBY_VISION_TYPE)).
352 forEach(c -> Arrays.stream(c.getCapabilitiesForType(DOLBY_VISION_TYPE).profileLevels).forEach(prof -> {
353
354 if (!dolbyProfileToCodecLookup.containsKey(prof.profile)) {
355 ArrayList<String> value = new ArrayList<>();
356 value.add(c.getCanonicalName());
357 dolbyProfileToCodecLookup.put(prof.profile, value);
358 } else {
359 Objects.requireNonNull(dolbyProfileToCodecLookup.get(prof.profile)).add(c.getCanonicalName());
360 }
361 }));
362
363 return dolbyProfileToCodecLookup;
364 }

371 public String printDolbyCodecSupport() {
372
373 final MediaCodecInfo.CodecProfileLevel reflexiveReference = new MediaCodecInfo.CodecProfileLevel();
374
375 StringBuilder builder = new StringBuilder();
376
377 HashMap<Integer, ArrayList<String>> profileMap = getDolbyCodecMapping();
378
379 Field[] codecInfoFields = MediaCodecInfo.CodecProfileLevel.class.getDeclaredFields();
380
381 //Get a list of supported Dolby profile names.
382 ArrayList<Field> fields = Arrays.stream(codecInfoFields).filter(f -> {
383 try {
384 return profileMap.containsKey(f.getInt(reflexiveReference)) && f.getName().startsWith(REFLEXIVE_DOLBY_PROFILE_LOOKUP);
385 } catch (IllegalAccessException e) {
386 e.printStackTrace();
387 }
388 return false;
389 }).collect(Collectors.toCollection(ArrayList::new));
390
391 for (int count = 0; count < fields.size(); count++) {
392 try {
393 builder.append(fields.get(count).getName()).append(" ").append(profileMap.get(fields.get(count).getInt(reflexiveReference))).append("\n");
394 } catch (IllegalAccessException e) {
395 e.printStackTrace();
396 }
397 }
398
399 return builder.toString();

410 private String filterCodecByFormat(MediaFormat format) {
411
412 //Build a format with the requirements a candidate codec will need to meet.
413 MediaFormat targetFormat = new MediaFormat();
414 targetFormat.setString(MediaFormat.KEY_MIME, format.getString(MediaFormat.KEY_MIME));
415 targetFormat.setInteger(MediaFormat.KEY_PROFILE, format.getInteger(MediaFormat.KEY_PROFILE));
416 targetFormat.setInteger(MediaFormat.KEY_LEVEL, format.getInteger(MediaFormat.KEY_LEVEL));
417 targetFormat.setInteger(MediaFormat.KEY_BIT_RATE, Integer.parseInt(retriever.extractMetadata(MediaMetadataRetriever.METADATA_KEY_BITRATE)));
418 MediaCodecList mediaCodecList = new MediaCodecList(MediaCodecList.REGULAR_CODECS);
419 String decoder = mediaCodecList.findDecoderForFormat(targetFormat);
420 return decoder;
421 }

445 @Override
446 public MediaExtractor configure(MediaCodec codec, Surface surface, BuilderCodecTemplate c, int transfer) throws NoCodecException {
447 MediaFormat format = this.getPlayableTrack().format;
448
449 this.decoderFormat = format;
450
451 String mime = format.getString(MediaFormat.KEY_MIME);
452
453 if ((decoderFormat.getString(MediaFormat.KEY_MIME).equals(MediaFormat.MIMETYPE_VIDEO_DOLBY_VISION)) &&
454 (decoderFormat.getInteger(MediaFormat.KEY_PROFILE) == MediaCodecInfo.CodecProfileLevel.DolbyVisionProfileDvheSt) &&
455 (outputFormat.equals(HEVC) || outputFormat.equals(AVC))) {
456 Log.e("CodecSelection", "configure: Dolby Vision Decoder Filter Enabled");
457 format.setInteger(MediaFormat.KEY_COLOR_TRANSFER_REQUEST, MediaFormat.COLOR_TRANSFER_SDR_VIDEO);
458 } else {
459 Log.e("CodecSelection", "configure: Not enabling Dolby Vision Filter");
460 }
461
462 if (mime.equals(MediaFormat.MIMETYPE_VIDEO_DOLBY_VISION)) {
463 c.setTransfer(BuilderCodecTemplate.TRANSFER_PARAMS[transfer]);
464 }
465
466 codec.configure(format, surface, null, 0);
467
468 return this.extractor;
469 }

476 @Override
477 public String getCodecName() throws IOException, NoCodecException {
478
479 String codecName;
480 do {
481 codecName = this.filterCodecByFormat(this.getPlayableTrack().format);
482 }

484
485 this.armExtractor(CodecBuilder.CodecType.VIDEO);
486
