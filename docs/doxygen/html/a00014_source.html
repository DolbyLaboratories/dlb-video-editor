72 public ImagePipeline(Size inputSize, Size outputSize, Surface outputSurface, boolean previewMode, int inputProfile, String encoderFormat) {
73
74 Log.d(TAG, "inputSize=" + inputSize + ", outputSize=" + outputSize + ", previewMode=" + previewMode + ", inputProfile=" + inputProfile + ", encoderFormat=" + encoderFormat);
75
76 // Configure dataspace
77 if (previewMode && inputProfile != MediaCodecInfo.CodecProfileLevel.DolbyVisionProfileDvheSt) {
78 this.dataspace = DataSpace.pack(DataSpace.STANDARD_BT709, DataSpace.TRANSFER_GAMMA2_2, DataSpace.RANGE_LIMITED);
79 } else {
80 if (encoderFormat.equals(DV_ME)) {
81 this.dataspace = DataSpace.pack(DataSpace.STANDARD_BT2020, DataSpace.TRANSFER_HLG, DataSpace.RANGE_LIMITED);
82 } else {
83 this.dataspace = DataSpace.pack(DataSpace.STANDARD_BT709, DataSpace.TRANSFER_GAMMA2_2, DataSpace.RANGE_LIMITED);
84 }
85 }
86
87 // Build ImageReader
88 this.reader = new ImageReader.Builder(inputSize.getWidth(), inputSize.getHeight())
89 .setMaxImages(MAX_IMAGES)
90 .setUsage(HardwareBuffer.USAGE_GPU_COLOR_OUTPUT)
91 .build();
92
93 // Build ImageWriter
94 int hwBufferFormat;
95 if (previewMode || encoderFormat.equals(DV_ME)) {
96 hwBufferFormat = HardwareBuffer.YCBCR_P010;
97 } else {
98 hwBufferFormat = HardwareBuffer.YCBCR_420_888;
99 }
100 this.writer = new ImageWriter.Builder(outputSurface)
101 .setWidthAndHeight(outputSize.getWidth(), outputSize.getHeight())
102 .setMaxImages(MAX_IMAGES)
103 .setDataSpace(this.dataspace)
104 .setHardwareBufferFormat(hwBufferFormat)
105 .build();
106
107 }
